```yaml
- uses: anthropics/claude-code-action@v1
  with:
    claude_code_oauth_token: ${{ secrets.CLAUDE_CODE_OAUTH_TOKEN }}
    settings: |
      {
        "permissions": {
          "allow": ["Read", "Write", "Edit", "MultiEdit", "Glob", "Grep", "Bash"]
        }
      }

      Activate and attach as https://github.com/usurobor/bumpt.
```
